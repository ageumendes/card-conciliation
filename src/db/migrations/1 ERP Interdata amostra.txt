-- 1) ERP (Interdata) - amostra
 SELECT FIRST 5
   ID, SALE_DATETIME, GROSS_AMOUNT, STATUS_RAW,
   CANON_SALE_DATE, CANON_METHOD_GROUP, CANON_GROSS_AMOUNT,
   CANON_TERMINAL_NO, AUTH_CODE, NSU, AUTH_NSU
 FROM T_INTERDATA_SALES
 WHERE CANON_SALE_DATE IS NOT NULL
 ORDER BY SALE_DATETIME DESC;

-- 2) CIELO - amostra
 SELECT FIRST 5
   ID, SALE_DATETIME, GROSS_AMOUNT, NET_AMOUNT, STATUS,
   CANON_SALE_DATE, CANON_METHOD_GROUP, CANON_GROSS_AMOUNT,
   CANON_TERMINAL_NO, CANON_AUTH_CODE, CANON_NSU
 FROM T_CIELO_SALES
 WHERE CANON_SALE_DATE IS NOT NULL
 ORDER BY SALE_DATETIME DESC;

-- 3) SIPAG - amostra
SELECT FIRST 5
   ID, SALE_DATETIME, GROSS_AMOUNT, STATUS,
   PAYMENT_METHOD, AUTH_NO, TRANSACTION_NO,
   CANON_SALE_DATE, CANON_METHOD_GROUP, CANON_GROSS_AMOUNT
 FROM T_SIPAG_SALES
 ORDER BY ID DESC;
